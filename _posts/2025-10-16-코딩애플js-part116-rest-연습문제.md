```javascript
//데이터마이닝 기능
/**알파벳들의 출현 개수를 세어주는 함수*/
function 글자세기(글자){
    let 결과 = {};
    [...글자].forEach(function(a){
        //만약에 결과에 a가 있으면 +1 없으면 a : 1로 집어넣으세요
        if(결과[a] > 0){
            결과[a] += 1
        }else{
            결과[a] = 1;
        }
    })
    console.log(결과)
}
글자세기('aacbbb')
```
