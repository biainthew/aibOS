function todoReducer(todos, action) {

        case 'INSERT':
            return todos.concat(action.todo);
        case 'REMOVE':
            return todos.filter((todo) => todo.id !== action.id);
        case 'TOGGLE':
            return todos.map((todo) =>

            );

            return todos;

    const onRemove = useCallback((id) => {
