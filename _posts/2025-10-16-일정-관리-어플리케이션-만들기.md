} from 'react-icons/md';

const TodoListItem = ({ todo, onRemove, onToggle }) => {
    const { id, text, checked } = todo;

                onClick={() => onToggle(id)}

            </div>
            <div
                className="remove"
                onClick={() => {
                    onRemove(id);
                }}

                <MdRemoveCircleOutline />
            </div>
        </div>
    );
};

export default TodoListItem;
