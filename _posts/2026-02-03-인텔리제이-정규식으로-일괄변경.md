1. `src="images/ico_instargram.png”`  →  `th:src="@{/portal/images/ico_youtube.png}"`

{% raw %}
```javascript
// Find
src="images/([^"]+)”

// Replace
th:src="@{/portal/images/$1}”
```
{% endraw %}

1. `id="check_03"`

{% raw %}
```javascript
id="[^"]+"
```
{% endraw %}

1. `href="main.html"`  →  `th:href="@{/portal/page/main}"`

{% raw %}
```javascript
// Find
href="([^"]+)\.html"

// Replace
th:href="@{/portal/page/$1}"
```
{% endraw %}

1. `src="../images/img/prv_request.png"`  →  `th:src="@{/portal/images/$1}"`

{% raw %}
```javascript
// Find
src="\.\.\/images\/[^\/]*\/([^"]+)"

// Replace
th:src="@{/portal/images/$1}"
```
{% endraw %}
