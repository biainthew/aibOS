### 정규식 조건에 넣을 때


```javascript
function solution(my_string) {
    let result = ''
    let arr = my_string.split('')
    let arr1 = arr.filter((v)=> v.match(/[1-9]/g));
    result = arr1.reduce((acc, cur) => {
        return Number(acc) + Number(cur)
    })
    return result;
}
```

### replace 조건 여러개 한번에 바꾸기


```javascript
function solution(rsp) {
    const mapObj = {
        0 : '5',
        2 : '0',
        5 : '2'
    };
    rsp = rsp.replace(/0|2|5/gi, function(matched){
        return mapObj[matched]
    })
    return rsp
}
```
