### **배열 추가 / 삭제 정리 (push, unshift, pop, shift, slice, filter, splice, length)**

- **배열** **추가** **[맨** **뒤****]** `push` : 배열의 맨 끝에 요소를 추가하고 **새로운 길이** 반환
- **배열** **추가** **[맨** **앞****]** `unshift`: 배열의 맨 앞에 요소를 추가하고 **새로운 길이** 반환
- **배열** **삭제** **[맨** **뒤****]** `pop` : 배열의 맨 끝 요소를 삭제하고 **제거된 요소** 반환
- **배열** **삭제** **[맨** **앞****]** `shift` : 배열의 맨 앞 요소를 삭제하고 **제거된 요소** 반환
- **배열** **삭제** **[해당 구간]** `slice(begin, end)` : `begin`부터 `end`까지 (`end` 미포함) 자름 / **원본배열 유지**
- **배열** **삭제** **[해당 요소]** `filter` : 해당 요소만 제외한 배열 반환
- **배열** **추가** **/** **삭제** `splice` : 기존 요소를 삭제 또는 교체하거나 새 요소를 추가하여 **배열의 내용 변경**
- **배열** **추가** **/** **삭제** `arr.length` : 배열의 길이를 변경하여 요소를 추가 / 삭제 가능

```javascript
let arr = ['1', '2', '3', '4'];

arr[arr.length] = '5';
console.log(arr); // ['1', '2', '3', '4', '5']


arr.length -= 1;
console.log(arr); // ['1', '2', '3', '4']


arr[5] = '6';
console.log(arr); // ['1', '2', '3', '4', undefined, '6']


let arr2 = arr;

arr = []; // arr는 새로운 비어있는 객체를 할당, arr2는 원래 배열을 참조

console.log(arr); // []
console.log(arr2); // ['1', '2', '3', '4', undefined, '6']


arr = ['1', '2', '3', '4'];
arr2 = arr;

arr.length = 0; // 원래 배열의 length를 0으로 변경

console.log(arr); // []
console.log(arr2); // []
```


### for of 사용 /  push 로 반복문 돌린 값 새 배열에 넣기


```javascript
function solution(array, commands) {
    let result = [];
    for (let [one, two, three] of commands){
        arrayN = array.slice(one-1, two)
        arrayN = arrayN.sort((a,b)=>a-b)
        result.push(arrayN[three-1])
        }
    return result
}
```


### 최빈값 구하기


```javascript
function solution(array) {
    let map = new Map();
    
    // Map 객체 초기화
    for(let i = 0; i <= Math.max(...array); i++){
        map.set(i, 0);    
    }
    
    // array 배열의 원소값과 일치하는 Map 객체의 key를 증가시킨다
    for(let i = 0; i < array.length; i++){
        map.set(array[i], map.get(array[i]) + 1);
    }
    
    // Map 객체의 value만 모아서 배열로 만든다
    let arr = Array.from(map.values());
    
    // 최대값 산출
    let max = Math.max(...arr);
    
    // 최대값이 하나만 있다면 max, 여러 개 중복된다면 -1 출력
    if(arr.indexOf(max) !== arr.lastIndexOf(max)){
        return -1;
    } else {
        return arr.indexOf(max);
    }
}
```
