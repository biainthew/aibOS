```javascript
function getNextZIndex() {
		//열려있는 모달
    const $modals = $("[class^='modal'][style*='display: flex']");
    let maxZIndex = 0;
		//모달 각각이
    $modals.each(function() {
				//$(this)의 z인덱스
        var zIndex = parseInt($(this).css("z-index"));
				//$(this)의 z인덱스가 0보다 크면
        if (zIndex > maxZIndex) {
						//maxZIndex에 대입
            maxZIndex = zIndex;
        }
    });
		//maxZIndex가 1보다 작으면 maxZIndex, 1보다 크면 +1 리턴
    return maxZIndex < 1 ? maxZIndex : maxZIndex + 1;
}
```
