var obj1 = { val: 10 };
function add4(obj, b) {
  obj.val += b;
}
console.log( obj1.val );
add4(obj1, 20);
console.log( obj1.val );


// 다시 순수 함수
var obj1 = { val: 10 };
function add5(obj, b) {
  return { val: obj.val + b }
}

console.log( obj1.val );
var obj2 = add5(obj1, 20);
console.log( obj1.val );
console.log( obj2.val );

  return f();

/* add_maker */

function add_maker(a) {
  return function(b) {
    return a + b;
  }
}

var add10 = add_maker(10);

console.log( add10(20) );

var add5 = add_maker(5);
var add15 = add_maker(15);

console.log( add5(10) );
console.log( add15(10) );

console.log( add10(20) );


function f4(f1, f2, f3) {
  return f3(f1() + f2());
}

console.log(
  f4(

### 순수함수


: 동일한 인자를 넣으면 항상 동일한 값을 리턴 / 평가시점이 중요하지 않음 / 외부 상태를 변화시키지 않음


### 일급함수


: 함수를 값으로 담을 수 있음 = 값을 변수로 담을 수 있음 / 인자로 함수를 받을 수 있음

- 순수함수를 많이 만들고 그 함수의 평가시점에 따라 어쩌구 로직이 …
