```javascript
<body>
    <div style="display : none">모달창0</div>
    <div style="display : none">모달창1</div>
    <div style="display : none">모달창2</div>

    <button>버튼0</button>
    <button>버튼1</button>
    <button>버튼2</button>
    <script>
        //1
        함수();
        function 함수() {
            console.log(안녕);
            let 안녕 = 'Hello!';
        }
        //undefined
        //선언 전에 엑세스 불가 에러
        
        //2
        함수();
        var 함수 = function(){
            console.log(안녕);
            var 안녕 = 'Hello!';
        }
        //undefined
        //함수는 함수가 아님 에러

        //3
        let a = 1;
        var 함수 = function(){
            a=2;
        }
        console.log(a);
        //1
        //함수를 만들기만 하고 실행하지 않았기 때문에

        // //4
        let a1 = 1;
        var b = 2;
        window.a1 = 3;
        window.b = 4;
        console.log(a1+b);
        // //5

        // //5
        for (let i = 1; i < 6; i++) { 
            setTimeout(function() {
                console.log(i);
            }, i*1000 ); 
        }
        //이유 : 1초 기다리는 동안 이미 반복문 다 돌아서 이미 i = 6 임

        //6
        var 버튼들 = document.querySelectorAll('button');
        var 모달창들 = document.querySelectorAll('div');

        for (let i = 0; i < 3; i++){
            버튼들[i].addEventListener('click', function(){
                모달창들[i].style.display = 'block';
            });
        }

    </script>
</body>
```
